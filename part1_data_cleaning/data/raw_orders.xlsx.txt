missing_ship_dates = df['ship_date'].isnull().sum()

print(f"Number of missing order_dates: {missing_order_dates}")
print(f"Number of missing ship_dates: {missing_ship_dates}")

# Identify invalid dates (where NaT resulted from coercion)
invalid_order_dates = df['order_date'].isna().sum() - missing_order_dates # should be 0 if only nulls become NaT
invalid_ship_dates = df['ship_date'].isna().sum() - missing_ship_dates # should be 0 if only nulls become NaT

print(f"Number of invalid order_dates (coerced to NaT): {df['order_date'].isna().sum()}")
print(f"Number of invalid ship_dates (coerced to NaT): {df['ship_date'].isna().sum()}")

# Identify instances where ship_date is earlier than order_date
invalid_shipping_records = df[df['ship_date'] < df['order_date']]

print(f"Number of records where ship_date is earlier than order_date: {num_invalid_shipping_records}")
